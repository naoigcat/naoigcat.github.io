## ヒープソートを使用する

ヒープソート (`heap sort`) は、配列を **二分ヒープ** の形に整え、その根に常に最大値が来る性質を利用して、根と末尾を入れ替えて確定させながらヒープを縮めていく **インプレース** の比較ソートである。

1.  **準備**: 配列をインデックス付きの完全二分木とみなす。親 `i` (`i >= 0`) に対し左の子は `2i+1`、右の子は `2i+2` になる。
2.  **構築**: 親の値が子の値以上という条件を満たすよう、葉に近い内部ノードから順に沈降して **最大ヒープ** にする。
3.  **抽出**: 根（位置 `0`）が全体の最大なので、それと **ヒープの末尾** を交換する。末尾側はもうソート済み領域として固定し、残り `[0 .. heapEnd-1]` だけをヒープとして見なす。
4.  **反復**: 新しい根がヒープ条件を満たすまで子と比較・交換して下へ沈める。ヒープ長が 1 になるまで手順 3〜4 を繰り返す。

```pseudocode
procedure heap_sort(A)
  n = length(A)
  for i from floor(n / 2) - 1 down to 0
    sift_down(A, i, n)
  for end from n - 1 down to 1
    swap(A[0], A[end])
    sift_down(A, 0, end)

procedure sift_down(A, start, heap_end)
  i = start
  while true
    l = 2 * i + 1
    r = 2 * i + 2
    largest = i
    if l < heap_end and A[l] > A[largest] then
      largest = l
    if r < heap_end and A[r] > A[largest] then
      largest = r
    if largest = i then
      break
    swap(A[i], A[largest])
    i = largest
```

最悪・平均・最良いずれも時間計算量は **O(n log n)** で、追加の配列を使わない標準実装では **空間計算量は O(1)**（再帰を使わない実装が一般的）。同一キーの相対順序を保たない **不安定** なソートであることが多い。

バブルソートの **O(n²)** と比べて大きな入力では優位になりやすく、クイックソートと異なり **最悪でも O(n log n)** に抑えられる反面、定数倍やキャッシュ効率では実装やデータに依存してクイックソートに劣ることもある。

<script>
(function () {
  var root = document.getElementById('heap-sort-demo');
  if (!root) return;
  var C = window.DemoSort;
  if (!C) return;

  function mountBars(container, values) {
    C.mountBars(container, values, 'heap-sort-demo__bar');
  }

  function generateSteps(initial) {
    var a = initial.slice();
    var steps = [];
    var n = a.length;

    function swapElements(i, j) {
      var t = a[i];
      a[i] = a[j];
      a[j] = t;
    }

    function siftDown(start, heapEnd, phaseLabel, sortedTail) {
      var i = start;
      while (true) {
        var l = 2 * i + 1;
        var r = 2 * i + 2;
        var largest = i;
        if (l < heapEnd) {
          steps.push({
            kind: 'compare',
            lo: largest,
            hi: l,
            arr: a.slice(),
            phase: phaseLabel,
            heapEnd: heapEnd,
            sortedFrom: sortedTail,
          });
          if (a[l] > a[largest]) largest = l;
        }
        if (r < heapEnd) {
          steps.push({
            kind: 'compare',
            lo: largest,
            hi: r,
            arr: a.slice(),
            phase: phaseLabel,
            heapEnd: heapEnd,
            sortedFrom: sortedTail,
          });
          if (a[r] > a[largest]) largest = r;
        }
        if (largest === i) break;
        swapElements(i, largest);
        steps.push({
          kind: 'swap',
          lo: i,
          hi: largest,
          arr: a.slice(),
          phase: phaseLabel,
          heapEnd: heapEnd,
          rootSwap: false,
          markSortedBeforeSwap:
            phaseLabel === 'extract' ? sortedTail : undefined,
          markSortedAfterSwap:
            phaseLabel === 'extract' ? sortedTail : undefined,
        });
        i = largest;
      }
    }

    if (n <= 1) {
      steps.push({ kind: 'done', arr: a.slice() });
      return steps;
    }

    steps.push({
      kind: 'caption',
      text: 'ヒープ構築: 内部ノードから sift-down して max-heap にする',
      arr: a.slice(),
      heapEnd: n,
      sortedFrom: n,
    });

    var bi;
    for (bi = Math.floor(n / 2) - 1; bi >= 0; bi--) {
      siftDown(bi, n, 'build', null);
    }

    steps.push({
      kind: 'caption',
      text: '構築完了。根（位置0）がヒープ内の最大値',
      arr: a.slice(),
      heapEnd: n,
      sortedFrom: n,
    });

    var end;
    for (end = n - 1; end > 0; end--) {
      steps.push({
        kind: 'caption',
        text:
          '抽出: 根とヒープ末尾（位置 ' +
          end +
          '）を入れ替え、末尾をソート済み領域へ',
        arr: a.slice(),
        heapEnd: end + 1,
        sortedFrom: end + 1,
      });
      swapElements(0, end);
      steps.push({
        kind: 'swap',
        lo: 0,
        hi: end,
        arr: a.slice(),
        phase: 'extract',
        heapEnd: end,
        rootSwap: true,
        markSortedBeforeSwap: end + 1,
        markSortedAfterSwap: end,
      });
      siftDown(0, end, 'extract', end);
      steps.push({
        kind: 'sorted_tick',
        arr: a.slice(),
        sortedFrom: end,
        heapEnd: end,
      });
    }

    steps.push({ kind: 'done', arr: a.slice() });
    return steps;
  }

  function clearRoles(container) {
    var nodes = container.children;
    for (var i = 0; i < nodes.length; i++) {
      nodes[i].removeAttribute('data-role');
    }
  }

  function paintSortedSuffix(container, sortedFrom) {
    clearRoles(container);
    if (
      sortedFrom != null &&
      sortedFrom < container.children.length
    ) {
      var i;
      for (i = sortedFrom; i < container.children.length; i++) {
        container.children[i].setAttribute('data-role', 'sorted');
      }
    }
  }

  function setComparePair(container, lo, hi) {
    var nodes = container.children;
    var i;
    for (i = 0; i < nodes.length; i++) {
      if (nodes[i].getAttribute('data-role') !== 'sorted') {
        nodes[i].removeAttribute('data-role');
      }
    }
    if (lo == null || hi == null) return;
    if (nodes[lo]) nodes[lo].setAttribute('data-role', 'compare');
    if (nodes[hi]) nodes[hi].setAttribute('data-role', 'compare');
  }

  var barsEl = root.querySelector('[data-hs="bars"]');
  var capEl = root.querySelector('[data-hs="caption"]');
  var btnShuffle = root.querySelector('[data-hs="shuffle"]');
  var btnPlay = root.querySelector('[data-hs="play"]');
  var btnPause = root.querySelector('[data-hs="pause"]');
  var btnStep = root.querySelector('[data-hs="step"]');

  var values = [5, 2, 8, 1, 9, 3, 6, 14, 4, 11, 7, 13, 10, 12, 15];
  var steps = [];
  var idx = 0;
  var playing = false;
  var cancelled = false;
  var busy = false;

  function syncButtons() {
    var atEnd = idx >= steps.length;
    btnPlay.disabled = playing || atEnd || busy;
    btnPause.disabled = !playing;
    btnStep.disabled = playing || atEnd || busy;
    btnShuffle.disabled = playing || busy;
  }

  function rebuild(v) {
    values = v;
    steps = generateSteps(values);
    idx = 0;
    cancelled = true;
    playing = false;
    busy = false;
    mountBars(barsEl, steps[0] ? steps[0].arr : values);
    clearRoles(barsEl);
    capEl.textContent =
      'ヒープソートのデモ（比較はオレンジ、交換は緑、ソート済み領域は紫）';
    syncButtons();
  }

  async function applyStepForward() {
    if (busy || idx >= steps.length) return;
    busy = true;
    syncButtons();
    try {
      var s = steps[idx];
      idx++;

      if (s.kind === 'caption') {
        mountBars(barsEl, s.arr);
        paintSortedSuffix(barsEl, s.sortedFrom);
        capEl.textContent = s.text;
        return;
      }

      if (s.kind === 'compare') {
        mountBars(barsEl, s.arr);
        paintSortedSuffix(barsEl, s.sortedFrom);
        setComparePair(barsEl, s.lo, s.hi);
        var ph = s.phase === 'build' ? '（ヒープ構築）' : '（抽出後の再ヒープ化）';
        capEl.textContent =
          'sift-down ' + ph + ': 位置 ' + s.lo + ' と ' + s.hi + ' を比較';
        return;
      }

      if (s.kind === 'swap') {
        if (s.phase === 'extract') {
          paintSortedSuffix(barsEl, s.markSortedBeforeSwap);
        } else {
          clearRoles(barsEl);
        }
        var swapNodes = barsEl.children;
        if (swapNodes[s.lo]) {
          swapNodes[s.lo].setAttribute('data-role', 'swap');
        }
        if (swapNodes[s.hi]) {
          swapNodes[s.hi].setAttribute('data-role', 'swap');
        }
        capEl.textContent = s.rootSwap
          ? '根と末尾を交換しています…'
          : 'ヒープ条件のため親子を交換しています…';
        await C.flipSwap(barsEl, s.lo, s.hi);
        if (s.phase === 'extract') {
          paintSortedSuffix(barsEl, s.markSortedAfterSwap);
        } else {
          clearRoles(barsEl);
        }
        capEl.textContent = '交換しました（位置 ' + s.lo + ' と ' + s.hi + '）';
        return;
      }

      if (s.kind === 'sorted_tick') {
        mountBars(barsEl, s.arr);
        paintSortedSuffix(barsEl, s.sortedFrom);
        capEl.textContent =
          '位置 ' + s.sortedFrom + ' 以降を確定（ソート済み）';
        return;
      }

      if (s.kind === 'done') {
        mountBars(barsEl, s.arr);
        clearRoles(barsEl);
        capEl.textContent = 'ソート完了';
      }
    } finally {
      busy = false;
      syncButtons();
    }
  }

  btnShuffle.addEventListener('click', function () {
    if (busy) return;
    rebuild(C.shuffleCopy(values));
  });

  btnStep.addEventListener('click', function () {
    applyStepForward();
  });

  btnPlay.addEventListener('click', async function () {
    playing = true;
    cancelled = false;
    syncButtons();
    while (!cancelled && idx < steps.length) {
      await applyStepForward();
      await C.wait(280);
    }
    playing = false;
    syncButtons();
  });

  btnPause.addEventListener('click', function () {
    cancelled = true;
    playing = false;
    syncButtons();
  });

  rebuild(values);
})();
</script>
</div>
