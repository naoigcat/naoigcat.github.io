## スローソートを使用する

スローソート (`slowsort`) は、区間を半分に分けて再帰的に整え、中央と末尾を比べて必要なら入れ替えたあと、末尾を除いた区間をもう一度整えるという手順を繰り返して昇順に整える。意図的に非効率な整列アルゴリズムとして知られる。

1.  **基底**: 部分配列 `A[lo .. hi]` で `lo ≥ hi` なら終了する。
2.  **分割**: `m = ⌊(lo + hi) / 2⌋` とする。
3.  **左半分**: `slowsort(A, lo, m)` を実行する。
4.  **右半分**: `slowsort(A, m + 1, hi)` を実行する。
5.  **中央と末尾**: `A[m] > A[hi]` なら入れ替える。
6.  **末尾を除く区間**: `slowsort(A, lo, hi - 1)` を実行する。
7.  **全体**: 配列全体 `[0 .. n-1]` に対して上記を適用する。

```pseudocode
procedure slowsort(A, lo, hi)
  if lo >= hi then
    return
  m = floor((lo + hi) / 2)
  slowsort(A, lo, m)
  slowsort(A, m + 1, hi)
  if A[m] > A[hi] then
    swap(A[m], A[hi])
  slowsort(A, lo, hi - 1)
```

最悪計算量は `O(n^{(log n) / 2})` で、多項式次数よりはるかに大きい（超多項式）。
再帰の深さは `O(log n)` だが、同じ要素を何度も扱い直すため実時間は非常に大きい。一般に不安定である。

<script>
window.DemoSort && DemoSort.boot('slow-sort-demo', function (root) {
  function rangePairs(lo, hi, role) {
    const pairs = [];
    for (let k = lo; k <= hi; k++) {
      pairs.push([k, role]);
    }
    return pairs;
  }

  function slowRange(a, lo, hi, steps) {
    steps.push({ kind: 'call_start', lo: lo, hi: hi, arr: a.slice() });
    if (lo >= hi) {
      steps.push({ kind: 'base_case', lo: lo, hi: hi, arr: a.slice() });
      return;
    }
    const m = Math.floor((lo + hi) / 2);
    steps.push({
      kind: 'phase',
      phase: 1,
      lo: lo,
      hi: hi,
      subLo: lo,
      subHi: m,
      arr: a.slice(),
    });
    slowRange(a, lo, m, steps);
    steps.push({
      kind: 'phase',
      phase: 2,
      lo: lo,
      hi: hi,
      subLo: m + 1,
      subHi: hi,
      arr: a.slice(),
    });
    slowRange(a, m + 1, hi, steps);
    steps.push({ kind: 'compare', lo: m, hi: hi, arr: a.slice() });
    if (a[m] > a[hi]) {
      const t = a[m];
      a[m] = a[hi];
      a[hi] = t;
      steps.push({ kind: 'swap', lo: m, hi: hi, arr: a.slice() });
    }
    steps.push({
      kind: 'phase',
      phase: 3,
      lo: lo,
      hi: hi,
      subLo: lo,
      subHi: hi - 1,
      arr: a.slice(),
    });
    slowRange(a, lo, hi - 1, steps);
  }

  function generateSteps(initial) {
    const a = initial.slice();
    const steps = [];
    if (a.length <= 1) {
      steps.push({ kind: 'done', arr: a.slice() });
      return steps;
    }
    slowRange(a, 0, a.length - 1, steps);
    steps.push({ kind: 'done', arr: a.slice() });
    return steps;
  }

  DemoSort.attachPlayback({
    root: root,
    dataAttr: 'data-slow',
    initialValues: [5, 2, 8, 1, 9, 3, 6, 14, 4, 11],
    initialCaption:
      'スローソートのデモ（対象区間は水色、中央と末尾の比較はオレンジ、交換は緑）',
    barClass: 'sort-demo__bar',
    generateSteps: generateSteps,
    applyStep: async function (api, s) {
      const barsEl = api.barsEl;
      if (s.kind === 'call_start') {
        api.mountBars(barsEl, s.arr);
        DemoSort.assignRoles(barsEl, rangePairs(s.lo, s.hi, 'range'));
        api.setCaption(
          '再帰呼び出し: 部分配列 位置 ' + s.lo + ' … ' + s.hi
        );
        return;
      }
      if (s.kind === 'compare') {
        api.mountBars(barsEl, s.arr);
        DemoSort.assignRoles(barsEl, [[s.lo, 'compare'], [s.hi, 'compare']]);
        api.setCaption('中央と末尾の比較: 位置 ' + s.lo + ' と ' + s.hi);
        return;
      }
      if (s.kind === 'swap') {
        DemoSort.assignRoles(barsEl, [[s.lo, 'swap'], [s.hi, 'swap']]);
        api.setCaption('交換しています…');
        await DemoSort.flipSwap(barsEl, s.lo, s.hi);
        DemoSort.clearRoles(barsEl);
        api.setCaption(
          '中央と末尾を交換しました（位置 ' + s.lo + ' と ' + s.hi + '）'
        );
        return;
      }
      if (s.kind === 'base_case') {
        api.mountBars(barsEl, s.arr);
        DemoSort.assignRoles(barsEl, rangePairs(s.lo, s.hi, 'range'));
        api.setCaption(
          '要素数 ' +
            (s.hi - s.lo + 1) +
            ' の区間 [' +
            s.lo +
            '…' +
            s.hi +
            '] — これ以上分割しない'
        );
        return;
      }
      if (s.kind === 'phase') {
        api.mountBars(barsEl, s.arr);
        DemoSort.assignRoles(
          barsEl,
          rangePairs(s.subLo, s.subHi, 'range')
        );
        const labels = {
          1: '第1段: 左半分 [' + s.subLo + '…' + s.subHi + '] を整列',
          2: '第2段: 右半分 [' + s.subLo + '…' + s.subHi + '] を整列',
          3: '第3段: 末尾を除く [' + s.subLo + '…' + s.subHi + '] を整列',
        };
        api.setCaption(labels[s.phase]);
        return;
      }
      if (s.kind === 'done') {
        api.mountBars(barsEl, s.arr);
        DemoSort.clearRoles(barsEl);
        api.setCaption('ソート完了');
      }
    },
    stepPauseMs: 280,
  });
});

## 類似アルゴリズムとの相違点

[ストゥージソート](/2026/07/20/sort-stooge.html)も再帰的に区間を三分割して整え直すが、スローソートは左半分・右半分・末尾を除いた区間という 3 段の再帰のあと、中央と末尾だけを一度比較する。

計算量は `O(n^{(log n) / 2})` と、ストゥージソートの `O(n^{log 3 / log 1.5})` より漸近的にさらに遅い。
