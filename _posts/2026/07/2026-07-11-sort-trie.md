## トライソートを使用する

トライソート（`trie sort`、接頭辞木ソート）は、各キーをトライ木に十進桁ごとに挿入し、子を `0..9` の順に深さ優先走査してキーを取り出す。

文字列の辞書順整列にトライ木を使う発想を、固定桁幅の整数キーへ拡張したものとみなせる。

[基数ソート](/2026/06/21/sort-radix.html) の MSD（最上位桁から）版と本質的に同じ区分付けを行うが、トライ木を明示的に構築してから走査する点が異なる。ビット列をキーとする場合は 2 分岐のビットトライ木となり、MSD 基数ソート／バイナリクイックソートと等価な区分付けになる。

1.  **桁幅の決定**: 配列の最大値 `max` から、最上位桁の重み `exp`（1, 10, 100, …）を求める。
2.  **挿入**: 各要素 `x` について、`(x / exp) % 10` から始め `exp` を 10 で割りながら子ノードへ降り、最終ノードの終端リストへ `x` を記録する（先頭ゼロ埋め相当）。
3.  **走査**: 根から各桁 `0..9` の順に子へ再帰し、終端リストの値を左から出力配列へ追加する。

```pseudocode
procedure trie_sort(A)
  if length(A) = 0 then return
  maxVal = maximum(A)
  exp = highest_digit_weight(maxVal)
  root = 空のトライ木ノード
  for each x in A
    insert x into root following digits from exp down to 1
  out = empty list
  dfs_collect(root, out)   // children 0..9 in order, then terminal values
  A = out
```

桁数 w なら `O(n · w)` となり、キーの大小比較を使わない非比較ソートである。

[ツリーソート](/2026/05/12/sort-tree.html) が二分探索木へ挿入して中順走査するのに対し、トライソートは桁ごとの `r` 分岐木を使う。値域が広く桁数が多いとノード数が増えやすいが、桁ごとの区分が MSD 基数ソートと同型であるため、分布が適度に散らばる整数列では線形に近い性能が期待できる。

<script>
window.DemoSort && DemoSort.boot('trie-sort-demo', function (root) {
  function maxExp(maxVal) {
    let exp = 1;
    while (exp * 10 <= maxVal) {
      exp *= 10;
    }
    return exp;
  }

  function digitName(exp) {
    if (exp === 1) {
      return '1の位';
    }
    if (exp === 10) {
      return '10の位';
    }
    if (exp === 100) {
      return '100の位';
    }
    return '桁の重み ' + exp;
  }

  function makeTrieNode() {
    return { children: {}, terminal: [] };
  }

  function trieInsert(node, value, exp, path, steps, arr) {
    if (exp === 0) {
      node.terminal.push(value);
      steps.push({
        kind: 'insert_done',
        value: value,
        path: path.slice(),
        arr: arr.slice(),
      });
      return;
    }
    const digit = Math.floor(value / exp) % 10;
    const nextPath = path.concat([digit]);
    steps.push({
      kind: 'insert_step',
      value: value,
      digit: digit,
      exp: exp,
      path: nextPath.slice(),
      arr: arr.slice(),
    });
    if (!node.children[digit]) {
      node.children[digit] = makeTrieNode();
    }
    trieInsert(node.children[digit], value, Math.floor(exp / 10), nextPath, steps, arr);
  }

  function trieCollect(node, digit, out, steps) {
    let d;
    for (d = 0; d < 10; d++) {
      if (node.children[d]) {
        steps.push({
          kind: 'traverse_enter',
          digit: d,
          out: out.slice(),
        });
        trieCollect(node.children[d], d, out, steps);
      }
    }
    let ti;
    for (ti = 0; ti < node.terminal.length; ti++) {
      out.push(node.terminal[ti]);
      steps.push({
        kind: 'traverse_emit',
        value: node.terminal[ti],
        out: out.slice(),
      });
    }
  }

  function formatPath(path) {
    if (!path.length) {
      return '（根）';
    }
    return path.join(' → ');
  }

  function mountOutputBars(container, values) {
    container.innerHTML = '';
    if (!values.length) {
      container.removeAttribute('role');
      container.removeAttribute('aria-label');
      return;
    }
    const min = Math.min.apply(null, values);
    const max = Math.max.apply(null, values);
    const span = Math.max(max - min, 1);
    container.setAttribute('role', 'list');
    container.setAttribute(
      'aria-label',
      'トライソートの出力配列。左から位置 0、1…の順です。'
    );
    let i;
    for (i = 0; i < values.length; i++) {
      const v = values[i];
      const bar = document.createElement('div');
      bar.className = 'sort-demo__bar';
      bar.setAttribute('role', 'listitem');
      bar.style.height = 28 + ((v - min) / span) * 92 + 'px';
      bar.setAttribute('title', String(v));
      bar.setAttribute(
        'aria-label',
        DemoSort.barAccessibilityLabel(i, String(v), null)
      );
      container.appendChild(bar);
    }
  }

  function generateSteps(initial) {
    const a = initial.slice();
    const steps = [];
    if (!a.length) {
      steps.push({ kind: 'done', arr: [] });
      return steps;
    }

    const maxVal = Math.max.apply(null, a);
    const exp = maxExp(maxVal);
    const trie = makeTrieNode();

    steps.push({
      kind: 'phase',
      phase: 'insert',
      arr: a.slice(),
      maxVal: maxVal,
      exp: exp,
    });

    let i;
    for (i = 0; i < a.length; i++) {
      steps.push({
        kind: 'insert_start',
        idx: i,
        value: a[i],
        arr: a.slice(),
      });
      trieInsert(trie, a[i], exp, [], steps, a);
    }

    const out = [];
    steps.push({
      kind: 'phase',
      phase: 'traverse',
      arr: a.slice(),
      out: out.slice(),
    });
    trieCollect(trie, -1, out, steps);

    steps.push({ kind: 'done', arr: out.slice() });
    return steps;
  }

  DemoSort.attachPlayback({
    root: root,
    dataAttr: 'data-trie',
    initialValues: [5, 2, 8, 1, 9, 3, 6, 14, 4, 11, 7, 13, 10, 12, 15],
    initialCaption:
      'トライソートのデモ（挿入は水色、走査出力は緑）',
    barClass: 'sort-demo__bar',
    generateSteps: generateSteps,
    afterRebuild: function (api) {
      api.mountBars(
        api.barsEl,
        api.steps[0] ? api.steps[0].arr : []
      );
    },
    applyStep: async function (api, s) {
      const barsEl = api.barsEl;
      if (s.kind === 'phase') {
        if (s.phase === 'traverse') {
          mountOutputBars(barsEl, s.out);
        } else {
          api.mountBars(barsEl, s.arr);
        }
        DemoSort.clearRoles(barsEl);
        if (s.phase === 'insert') {
          api.setCaption(
            'フェーズ1: 各値をトライ木へ挿入（最大値 ' +
              s.maxVal +
              '、最上位桁 ' +
              digitName(s.exp) +
              ' から）'
          );
        } else {
          api.setCaption(
            'フェーズ2: 子 0..9 の順に深さ優先走査して出力配列を構築します'
          );
        }
        return;
      }
      if (s.kind === 'insert_start') {
        api.mountBars(barsEl, s.arr);
        DemoSort.assignRoles(barsEl, [[s.idx, 'cursor']]);
        api.setCaption(
          '挿入: 位置 ' + s.idx + ' の値 ' + s.value + ' をトライ木へ追加します'
        );
        return;
      }
      if (s.kind === 'insert_step') {
        api.mountBars(barsEl, s.arr);
        DemoSort.clearRoles(barsEl);
        api.setCaption(
          '値 ' +
            s.value +
            ' の ' +
            digitName(s.exp) +
            ' = ' +
            s.digit +
            '（経路: ' +
            formatPath(s.path) +
            '）'
        );
        return;
      }
      if (s.kind === 'insert_done') {
        api.mountBars(barsEl, s.arr);
        DemoSort.clearRoles(barsEl);
        api.setCaption(
          '値 ' +
            s.value +
            ' を終端ノードへ記録（経路: ' +
            formatPath(s.path) +
            '）'
        );
        return;
      }
      if (s.kind === 'traverse_enter') {
        mountOutputBars(barsEl, s.out);
        DemoSort.clearRoles(barsEl);
        api.setCaption('走査: 桁 ' + s.digit + ' の子へ進みます');
        return;
      }
      if (s.kind === 'traverse_emit') {
        mountOutputBars(barsEl, s.out);
        DemoSort.assignRoles(barsEl, [[s.out.length - 1, 'write']]);
        api.setCaption(
          '出力: 値 ' +
            s.value +
            ' を位置 ' +
            (s.out.length - 1) +
            ' へ追加'
        );
        return;
      }
      if (s.kind === 'done') {
        mountOutputBars(barsEl, s.arr);
        DemoSort.clearRoles(barsEl);
        api.setCaption('ソート完了');
      }
    },
    stepPauseMs: 280,
  });
});
</script>

## 基数ソートとの関係

MSD 基数ソートは、各桁パスでカウンティングと配置を行い、部分配列を再帰的に処理する。トライソートは同じ桁区分を木構造として保持し、構築後に一度走査する。

| 観点 | トライソート | MSD 基数ソート |
| --- | --- | --- |
| データ構造 | 明示的なトライ木 | カウンティング配列＋再帰 |
| 走査 | 構築後に DFS 1 回 | 桁ごとに配置と再帰 |
| 追加メモリ | ノードポインタ分 | カウンティング用 `O(r)` |
| 典型用途 | 文字列・可変長キー | 固定桁整数・文字列 |

Burstsort などは文字列整列でトライ木とバケットを組み合わせ、大規模データ向けに最適化している。
